@def(structs)
struct Schedule;

typedef void (* act_Callback)(
struct Schedule *schedule,
void *context
);

struct Action {
struct lst_Node @priv(node);
act_Callback @priv(callback);
void *@priv(context);
#if CONFIG_WITH_MAGIC
unsigned @priv(magic);
#endif
};

@expand(functions)
@end(structs)
@def(functions)
static inline bool isAction(
const struct Action *a
) {
if (! a) { return false; }
#if CONFIG_WITH_MAGIC
if (a->@priv(magic) !=
@magic(action)
) {
return false;
}
#endif
return true;
}
@end(functions)
@add(functions)
static inline
struct Action *initAction(
struct Action *a,
act_Callback cb,
void *ctx
) {
if (! a) { return NULL; }
if (! cb) { return NULL; }
a->@priv(callback) = cb;
a->@priv(context) = ctx;
#if CONFIG_WITH_MAGIC
a->@priv(magic) = @magic(action);
#endif
return a;
}
@end(functions)
@add(functions)
bool invokeAction(
struct Schedule *s,
struct Action *a
)
#if act_IMPL
{
if (! s) { return false; }
if (! isAction(a)) { return false; }
act_Callback cb = a->@priv(callback);
if (! cb) { return false; }
cb(s, a->@priv(context));
return true;
}
#else
;
#endif
@end(functions)
